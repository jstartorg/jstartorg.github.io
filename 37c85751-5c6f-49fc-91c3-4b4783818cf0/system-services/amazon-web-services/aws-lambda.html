Description

Lambda is a "serverless" FaaS service that allows us to create single/simple functions written in [Python](https://en.wikipedia.org/wiki/Python_(programming_language) or Javascript that run when they are called (and are billed for their run time) but are otherwise idle. That means we don't have to pay for a server to stay running to server requests, but we can still develop in normal languages.

import sys
import os
import re
import json
import yaml
import psycopg2
import datetime

#Lambda handler
def lambda_handler(event, context):
    #No exception handling is used on the connection attempt because we want failed connection 
    #attempts to fail the function so that messages go back into the queue
    conn = psycopg2.connect(host=os.environ['rds_host'], port=os.environ['rds_port'], dbname=os.environ['rds_dbname'], 
                            user=os.environ['rds_username'], password=os.environ['rds_password'], connect_timeout=2)
    conn.autocommit = True
    cur = conn.cursor()
    for record in event['Records']:
        # process JSON text
        jsonText = record.get('body').encode('utf-8').decode('unicode_escape')
        jsonText = jsonText.replace("'\"\"","\"").replace("W/\"datetime\'","") # fix etag from Cognito Forms
        jsonText = re.sub('data-group-id="(\w+)"','data-group-id=\\"\1\\"', jsonText) # fix html tags from Easy Generator
        # process IP address
        ipAddress = record.get('messageAttributes').get('ipaddress').get('stringValue') if record.get('messageAttributes').get('ipaddress') is not None and record.get('messageAttributes').get('ipaddress').get('stringValue') is not None else ''
        try:
            # trying to parse JSON
            try:
                payload = json.loads(jsonText)
            except:
                raise
            try:
                if payload is not None:
                    payload = yaml.load(jsonText, yaml.Loader)
            except:
                raise
            # sometimes the payload is itself already an array, sometimes it is a single record
            if not isinstance(payload,list):
                payloads = [payload]
            else:
                payloads = payload

            for p in payloads:
                origin = 'sqs'
                emailId = 'TBD'
                if jsonText != '{"MessageBody": {}}':
                    # process email_id
                    if p.get('actor') is not None:
                        if p.get('actor').get('mbox') is not None:
                            emailId = p['actor']['mbox'].replace('mailto:','')
                        elif p.get('actor').get('account') is not None:
                            if p.get('actor').get('account').get('name') is not None:
                                emailId = p.get('actor').get('account').get('name').replace('mailto:','')
                    elif p.get('Email') is not None:
                        emailId = p.get('Email')
                    elif p.get('email') is not None:
                        emailId = p.get('email')
                    elif p.get('SalesforceId') is not None:
                        emailId = p.get('SalesforceId')

                    # process origin
                    if any(re.findall(r'easygenerator|linkedinlearning|lyndaCourse', repr(p), re.IGNORECASE)):
                        origin = 'course'
                    elif p.get('origin') is not None:
                        origin = p.get('origin')
                    elif p.get('Origin') is not None:
                        origin = p.get('Origin')
                    elif p.get('context') is not None and p.get('context').get('origin') is not None:
                        origin = p.get('context').get('origin')
                    elif p.get('Form') is not None and p.get('Form').get('InternalName') is not None:
                        origin = p.get('Form').get('InternalName')
                try:
                    cur.execute('insert into "warehouse" ("data", "email_id", "origin","ip_address") values (%s, %s, %s, %s)', 
                            (json.dumps(p, separators=(',', ':')), emailId, origin, ipAddress))
                except psycopg2.IntegrityError:
                    raise
                    #Preventing duplicate inserts and ignoring failed attempts allows this function to be idempotent
                    # print('PostgreSQL integrity error: ' + str(err)
        except Exception as err:
            try:
                cur.execute('insert into "warehouse" ("rawtext", "email_id","origin","ip_address") values (%s, %s, %s, %s)', 
                    (jsonText, 'unknown', 'json_error', ipAddress))
            except Exception as writeerr:
                #Preventing duplicate inserts and ignoring failed attempts allows this function to be idempotent
                print('PostgreSQL error: ' + str(writeerr))
    cur.close()
    conn.close()
