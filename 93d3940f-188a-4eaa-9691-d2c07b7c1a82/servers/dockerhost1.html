####### Windmill.dev setup
  windmill_server:
    container_name: windmill_server
    image: ghcr.io/windmill-labs/windmill:main
    deploy:
      replicas: 1
    restart: unless-stopped
    expose:
      - 8000
    environment:
      - DATABASE_URL=postgres://postgres:${POSTGRES_LOCAL_PASSWORD}@postgres_local/pg_local?sslmode=disable
      - BASE_URL=${WINDMILL_BASE_URL}
      - RUST_LOG=info
      ## You can set the number of workers to > 0 and not need any separate worker service
      - NUM_WORKERS=0
      - DISABLE_SERVER=false
      - METRICS_ADDR=false
      - PIP_LOCAL_DEPENDENCIES=smbclient,spnego
      - ZOMBIE_JOB_TIMEOUT=120
    depends_on:
      postgres_local:
        condition: service_healthy
    volumes:
     - /var/docker/windmill/oauth.json/:/usr/src/app/oauth.json

  windmill_worker:
    #container_name: windmill_worker
    image: ghcr.io/windmill-labs/windmill:main
    deploy:
      replicas: 5
      resources:
        limits:
          cpus: "2"
          memory: 2048M
    restart: unless-stopped
    environment:
      - DATABASE_URL=postgres://postgres:${POSTGRES_LOCAL_PASSWORD}@postgres_local/pg_local?sslmode=disable
      - BASE_URL=${WINDMILL_BASE_URL}
      - BASE_INTERNAL_URL=http://windmill_server:8000
      - NUM_WORKERS=1
      - MODE=worker
      - WORKER_GROUP=default
      - DISABLE_SERVER=true
      - KEEP_JOB_DIR=false
      - DENO_PATH=/usr/bin/deno
      - PYTHON_PATH=/usr/local/bin/python3
      - METRICS_ADDR=false
      - PIP_LOCAL_DEPENDENCIES=smbclient,spnego
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock
      - /var/docker/windmill/worker_dependency_cache:/tmp/windmill/cache
    depends_on:
      postgres_local:
        condition: service_healthy
  windmill_worker_native:
